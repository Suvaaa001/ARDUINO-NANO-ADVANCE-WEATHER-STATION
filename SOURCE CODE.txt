unsigned long previousMillis = 0;
const long interval = 3000;
int displayState = 0;

void setup() {
  Serial.begin(9600);
  display.begin(SH1106_SWITCHCAPVCC, 0x3C);
  display.clearDisplay();
  display.display();
  dht.begin();
  
  // Show "Weather Station" with animation
   showProgressBar();
}

void loop() {
  unsigned long currentMillis = millis();
  
  if (currentMillis - previousMillis >= interval) {
    previousMillis = currentMillis;
    
    displayState++;
    if (displayState > 3) {
      displayState = 1;
    }
    
    display.clearDisplay();
    
    switch (displayState) {
      case 1: // Temperature
        displayTemperature();
        break;
      case 2: // Humidity
        displayHumidity();
        break;
      case 3: // AQI
        displayAQI();
        break;
    }
    
    display.display();
  }
}

void showProgressBar() {
  display.clearDisplay();
  
  // Display "Weather Station" text
  display.setTextSize(1.5);
  display.setTextColor(WHITE);
  String title = "Weather Station \n      starting";
  int16_t x1, y1;
  uint16_t w, h;
  display.getTextBounds(title, 0, 0, &x1, &y1, &w, &h);
  display.setCursor((display.width() - w)/2, 15);
  display.print(title);
  
  // Draw progress bar
  int barHeight = 8;
  int barY = display.height() - barHeight - 10;
  display.drawRect(10, barY, display.width()-20, barHeight, WHITE);
  
  // Animate progress bar
  for(int i = 0; i <= 100; i++) {
    display.fillRect(12, barY+2, map(i, 0, 100, 0, display.width()-24), barHeight-4, WHITE);
    display.display();
    delay(30);
  }
  
  delay(500);
  display.clearDisplay();
  display.display();
}

void displayTemperature() {
  float t = dht.readTemperature();
  
  // Header
  display.setTextSize(1);
  display.setCursor(0, 0);
  display.println("Temperature");
  display.drawLine(0, 10, display.width(), 10, WHITE);
  
  // Big icon left-aligned
  display.drawBitmap(0, 0, thermometer_icon, 128, 64, WHITE);
  
  // Big value right-aligned
  display.setTextSize(2);
  if (isnan(t)) {
    display.setCursor(60, 25);
    display.println("Error");
  } else {
    String tempStr = String(t, 1) + (char)247 + "C";
    // Calculate position to right-align
    int16_t x1, y1;
    uint16_t w, h;
    display.getTextBounds(tempStr, 0, 0, &x1, &y1, &w, &h);
    display.setCursor(display.width() - w - 5, 25);
    display.print(tempStr);
  }
}

void displayHumidity() {
  float h = dht.readHumidity();
  
  // Header
  display.setTextSize(1);
  display.setCursor(0, 0);
  display.println("Humidity");
  display.drawLine(0, 10, display.width(), 10, WHITE);
  
  // Big icon left-aligned
  display.drawBitmap(0, 0, humidity_icon, 128, 64, WHITE);
  
  // Big value right-aligned
  display.setTextSize(3);
  if (isnan(h)) {
    display.setCursor(60, 25);
    display.println("Error");
  } else {
    String humStr = String(h, 0) + "%";
    // Calculate position to right-align
    int16_t x1, y1;
    uint16_t w, h;
    display.getTextBounds(humStr, 0, 0, &x1, &y1, &w, &h);
    display.setCursor(display.width() - w - 5, 25);
    display.print(humStr);
  }
}

void displayAQI() {
  int sensorValue = analogRead(MQ135_PIN);
  int aqi = map(sensorValue, 0, 1023, 0, 500);
  
  // Header
  display.setTextSize(1);
  display.setCursor(0, 0);
  display.println("Air Quality");
  display.drawLine(0, 10, display.width(), 10, WHITE);
  
  // Big icon left-aligned
  display.drawBitmap(0, 0, aqi_icon, 128, 64, WHITE);
  
  // Big value right-aligned
  display.setTextSize(2);
  String aqiStr = String(aqi) + " PPM";
  // Calculate position to right-align
  int16_t x1, y1;
  uint16_t w, h;
  display.getTextBounds(aqiStr, 0, 0, &x1, &y1, &w, &h);
  display.setCursor(display.width() - w - 5, 25);
  display.print(aqiStr);
  
  // Visual indicator
  display.fillRect(0, 50, map(aqi, 0, 500, 0, 128), 5, aqi < 150 ? WHITE : (aqi < 300 ? WHITE : WHITE));
}
